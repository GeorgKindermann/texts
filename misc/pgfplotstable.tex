\newcommand*{\kreise}[1]{\begin{tikzpicture}[scale=0.15]%
    \draw (0,0) circle (1);
    \fill[fill opacity=1,fill=black] (0,0) circle (#1/100);
  \end{tikzpicture}}

\newcolumntype{x}[1]{>{\hspace{1cm}}p{#1}\kreise{##1}}

\end{table}
